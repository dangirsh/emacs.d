** Function : update packages once a week

Update all packages every 7 days

#+begin_src emacs-lisp
(defun pl/el-get-update-once-a-week()
  (interactive)
  (if (= 0 (% 7 (string-to-number (format-time-string "%d"))))
    (progn
      (el-get-self-update)
      (el-get-update-all))))
